<Note>
  Starting with v3.7.0, OpenAPI requires **signature v2** (`X-Nonce`, Base64 signature; SHA-256 of non-empty bodies). See [Authentication](/authentication) for the full rules.
</Note>

MC OpenAPI does not currently ship an official SDK. Because every request follows the same signing rules ([Authentication](/authentication)), a thin client that wraps HMAC-SHA256 signature v2 plus a standard HTTP library is all you need.

The reference snippets below are production-ready starting points. Copy, adapt, and integrate them into your codebase.

## Python

```python
"""mc_client.py — minimal MC OpenAPI client (signature v2)."""
import base64, hmac, hashlib, json, time, uuid, os
from urllib.parse import urlencode
import requests


class MCClient:
    def __init__(self, access_key: str, secret_key: str,
                 base_url: str = "https://gateway.mcconnects.com"):
        self.access_key = access_key
        self.secret_key = secret_key.encode()
        self.base_url = base_url.rstrip("/")

    def _sign(self, method: str, path: str, query: str, body_bytes: bytes, ts: str, nonce: str) -> str:
        parts = [method.upper(), path]
        if query:
            parts.append(query)
        signed_headers = "&".join(
            f"{k}={v}"
            for k, v in sorted(
                {
                    "x-access-key": self.access_key,
                    "x-nonce": nonce,
                    "x-timestamp": ts,
                }.items()
            )
        )
        parts.extend([signed_headers, ts, nonce])
        if body_bytes and method.upper() in {"POST", "PUT", "PATCH"}:
            parts.append(hashlib.sha256(body_bytes).hexdigest())
        string_to_sign = "\n".join(parts) + "\n"
        return base64.b64encode(
            hmac.new(self.secret_key, string_to_sign.encode(), hashlib.sha256).digest()
        ).decode()

    def request(self, method: str, path: str, *, params=None, json_body=None):
        query = urlencode(sorted((params or {}).items()))
        body_bytes = (
            json.dumps(json_body, separators=(",", ":")).encode()
            if json_body is not None
            else b""
        )
        ts = str(int(time.time() * 1000))
        nonce = str(uuid.uuid4())
        signature = self._sign(method.upper(), path, query, body_bytes, ts, nonce)

        url = self.base_url + path + (f"?{query}" if query else "")
        headers = {
            "X-Access-Key": self.access_key,
            "X-Signature": signature,
            "X-Timestamp": ts,
            "X-Nonce": nonce,
            "X-Signature-Version": "2",
            "Accept": "application/json",
        }
        if body_bytes:
            headers["Content-Type"] = "application/json"

        r = requests.request(method, url, headers=headers, data=body_bytes or None, timeout=15)
        r.raise_for_status()
        return r.json()


# Usage
client = MCClient(os.environ["MC_ACCESS_KEY"], os.environ["MC_SECRET_KEY"])
rates = client.request("GET", "/openapi/v1/mc-account/exchange-rate/query/usd")
print(rates)
```

## Node.js

```javascript
// mcClient.js — minimal MC OpenAPI client (signature v2, Node.js 18+, ESM).
import crypto from "crypto";

export class MCClient {
  constructor({ accessKey, secretKey, baseUrl = "https://gateway.mcconnects.com" }) {
    this.accessKey = accessKey;
    this.secretKey = secretKey;
    this.baseUrl = baseUrl.replace(/\/$/, "");
  }

  #sign(method, path, query, bodyBuffer, ts, nonce) {
    const parts = [method.toUpperCase(), path];
    if (query) parts.push(query);

    const signHeaders = {
      "x-access-key": this.accessKey,
      "x-nonce": nonce,
      "x-timestamp": ts,
    };
    parts.push(
      Object.keys(signHeaders)
        .sort()
        .map((k) => `${k}=${signHeaders[k]}`)
        .join("&"),
      ts,
      nonce
    );

    const body = bodyBuffer || Buffer.alloc(0);
    if (body.length > 0 && ["POST", "PUT", "PATCH"].includes(method.toUpperCase())) {
      parts.push(crypto.createHash("sha256").update(body).digest("hex"));
    }

    const stringToSign = parts.join("\n") + "\n";
    return crypto.createHmac("sha256", this.secretKey).update(stringToSign).digest("base64");
  }

  async request(method, path, { params = {}, jsonBody } = {}) {
    const query = new URLSearchParams(
      Object.entries(params).sort(([a], [b]) => a.localeCompare(b))
    ).toString();
    const bodyBuffer =
      jsonBody !== undefined ? Buffer.from(JSON.stringify(jsonBody)) : Buffer.alloc(0);
    const ts = Date.now().toString();
    const nonce = crypto.randomUUID();
    const signature = this.#sign(method.toUpperCase(), path, query, bodyBuffer, ts, nonce);

    const url = this.baseUrl + path + (query ? `?${query}` : "");
    const headers = {
      "X-Access-Key": this.accessKey,
      "X-Signature": signature,
      "X-Timestamp": ts,
      "X-Nonce": nonce,
      "X-Signature-Version": "2",
      Accept: "application/json",
    };
    if (bodyBuffer.length > 0) headers["Content-Type"] = "application/json";

    const res = await fetch(url, {
      method,
      headers,
      body: bodyBuffer.length > 0 ? bodyBuffer : undefined,
    });
    if (!res.ok) throw new Error(`HTTP ${res.status}: ${await res.text()}`);
    return res.json();
  }
}

// Usage
const client = new MCClient({
  accessKey: process.env.MC_ACCESS_KEY,
  secretKey: process.env.MC_SECRET_KEY,
});
const rates = await client.request("GET", "/openapi/v1/mc-account/exchange-rate/query/usd");
console.log(rates);
```

## Java

```java
// MCClient.java — minimal MC OpenAPI client (signature v2, Java 11+).
import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.nio.charset.StandardCharsets;
import java.security.MessageDigest;
import java.time.Duration;
import java.util.Base64;
import java.util.Map;
import java.util.TreeMap;
import java.util.UUID;
import java.util.stream.Collectors;
import javax.crypto.Mac;
import javax.crypto.spec.SecretKeySpec;

public class MCClient {
    private final String accessKey;
    private final byte[] secretKey;
    private final String baseUrl;
    private final HttpClient http = HttpClient.newBuilder()
            .connectTimeout(Duration.ofSeconds(10)).build();

    public MCClient(String accessKey, String secretKey, String baseUrl) {
        this.accessKey = accessKey;
        this.secretKey = secretKey.getBytes(StandardCharsets.UTF_8);
        this.baseUrl = baseUrl.replaceAll("/$", "");
    }

    private String sign(String method, String path, String query, byte[] bodyBytes, String ts, String nonce) {
        try {
            StringBuilder sb = new StringBuilder();
            sb.append(method.toUpperCase()).append("\n").append(path).append("\n");
            if (query != null && !query.isEmpty()) {
                sb.append(query).append("\n");
            }

            TreeMap<String, String> signHeaders = new TreeMap<>();
            signHeaders.put("x-access-key", accessKey);
            signHeaders.put("x-nonce", nonce);
            signHeaders.put("x-timestamp", ts);
            sb.append(signHeaders.entrySet().stream()
                    .map(e -> e.getKey() + "=" + e.getValue())
                    .collect(Collectors.joining("&"))).append("\n");
            sb.append(ts).append("\n").append(nonce).append("\n");

            if (bodyBytes != null && bodyBytes.length > 0 && isWriteMethod(method)) {
                sb.append(sha256Hex(bodyBytes)).append("\n");
            }

            Mac mac = Mac.getInstance("HmacSHA256");
            mac.init(new SecretKeySpec(secretKey, "HmacSHA256"));
            return Base64.getEncoder().encodeToString(
                    mac.doFinal(sb.toString().getBytes(StandardCharsets.UTF_8)));
        } catch (Exception e) {
            throw new RuntimeException(e);
        }
    }

    private static boolean isWriteMethod(String method) {
        String m = method.toUpperCase();
        return "POST".equals(m) || "PUT".equals(m) || "PATCH".equals(m);
    }

    private static String sha256Hex(byte[] bytes) throws Exception {
        byte[] hash = MessageDigest.getInstance("SHA-256").digest(bytes);
        StringBuilder hex = new StringBuilder(hash.length * 2);
        for (byte b : hash) hex.append(String.format("%02x", b));
        return hex.toString();
    }

    public HttpResponse<String> request(String method, String path,
                                        Map<String, String> params, String body) throws Exception {
        StringBuilder qb = new StringBuilder();
        if (params != null) {
            for (Map.Entry<String, String> e : new TreeMap<>(params).entrySet()) {
                if (qb.length() > 0) qb.append('&');
                qb.append(e.getKey()).append('=').append(e.getValue());
            }
        }
        String query = qb.toString();
        byte[] bodyBytes = body == null ? new byte[0] : body.getBytes(StandardCharsets.UTF_8);
        String ts = String.valueOf(System.currentTimeMillis());
        String nonce = UUID.randomUUID().toString();
        String signature = sign(method, path, query, bodyBytes, ts, nonce);

        String url = baseUrl + path + (query.isEmpty() ? "" : "?" + query);
        HttpRequest.Builder b = HttpRequest.newBuilder(URI.create(url))
                .header("X-Access-Key", accessKey)
                .header("X-Signature", signature)
                .header("X-Timestamp", ts)
                .header("X-Nonce", nonce)
                .header("X-Signature-Version", "2")
                .header("Accept", "application/json");
        if (bodyBytes.length > 0) b.header("Content-Type", "application/json");
        b.method(method,
                bodyBytes.length == 0
                        ? HttpRequest.BodyPublishers.noBody()
                        : HttpRequest.BodyPublishers.ofByteArray(bodyBytes));

        return http.send(b.build(), HttpResponse.BodyHandlers.ofString());
    }
}
```

## Go

```go
// mcclient.go — minimal MC OpenAPI client (signature v2).
package mc

import (
    "crypto/hmac"
    "crypto/rand"
    "crypto/sha256"
    "encoding/base64"
    "encoding/hex"
    "fmt"
    "io"
    "net/http"
    "net/url"
    "sort"
    "strings"
    "time"
)

type Client struct {
    AccessKey string
    SecretKey []byte
    BaseURL   string
    HTTP      *http.Client
}

func New(accessKey, secretKey, baseURL string) *Client {
    return &Client{
        AccessKey: accessKey,
        SecretKey: []byte(secretKey),
        BaseURL:   strings.TrimRight(baseURL, "/"),
        HTTP:      &http.Client{Timeout: 15 * time.Second},
    }
}

func newNonce() string {
    b := make([]byte, 16)
    rand.Read(b)
    return hex.EncodeToString(b)
}

func (c *Client) sign(method, path, query, body, ts, nonce string) string {
    parts := []string{strings.ToUpper(method), path}
    if query != "" {
        parts = append(parts, query)
    }

    headers := map[string]string{
        "x-access-key": c.AccessKey,
        "x-nonce":      nonce,
        "x-timestamp":  ts,
    }
    keys := make([]string, 0, len(headers))
    for k := range headers {
        keys = append(keys, k)
    }
    sort.Strings(keys)
    var hb strings.Builder
    for i, k := range keys {
        if i > 0 {
            hb.WriteByte('&')
        }
        hb.WriteString(k)
        hb.WriteByte('=')
        hb.WriteString(headers[k])
    }
    parts = append(parts, hb.String(), ts, nonce)

    m := strings.ToUpper(method)
    if body != "" && (m == "POST" || m == "PUT" || m == "PATCH") {
        h := sha256.Sum256([]byte(body))
        parts = append(parts, hex.EncodeToString(h[:]))
    }

    canonical := strings.Join(parts, "\n") + "\n"
    mac := hmac.New(sha256.New, c.SecretKey)
    mac.Write([]byte(canonical))
    return base64.StdEncoding.EncodeToString(mac.Sum(nil))
}

func (c *Client) Do(method, path string, params url.Values, body string) ([]byte, error) {
    keys := make([]string, 0, len(params))
    for k := range params {
        keys = append(keys, k)
    }
    sort.Strings(keys)
    parts := make([]string, 0, len(keys))
    for _, k := range keys {
        parts = append(parts, k+"="+params.Get(k))
    }
    query := strings.Join(parts, "&")

    ts := fmt.Sprintf("%d", time.Now().UnixMilli())
    nonce := newNonce()
    signature := c.sign(method, path, query, body, ts, nonce)

    full := c.BaseURL + path
    if query != "" {
        full += "?" + query
    }
    req, err := http.NewRequest(method, full, strings.NewReader(body))
    if err != nil {
        return nil, err
    }
    req.Header.Set("X-Access-Key", c.AccessKey)
    req.Header.Set("X-Signature", signature)
    req.Header.Set("X-Timestamp", ts)
    req.Header.Set("X-Nonce", nonce)
    req.Header.Set("X-Signature-Version", "2")
    req.Header.Set("Accept", "application/json")
    if body != "" {
        req.Header.Set("Content-Type", "application/json")
    }

    resp, err := c.HTTP.Do(req)
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()
    return io.ReadAll(resp.Body)
}
```

<Tip>
  All snippets sign requests using **signature v2** as described in [Authentication](/authentication). If a signature is rejected (HTTP 401, message `签名无效`), print your string to sign and compare it byte-for-byte with the reference implementation.
</Tip>
